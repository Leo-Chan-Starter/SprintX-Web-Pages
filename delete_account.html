// deno-lint-ignore-file no-explicit-any // 此 Edge Function 部署於 Supabase，命名為 delete_account import { serve } from "https://deno.land/std@0.168.0/http/server.ts"; import { createClient } from "https://esm.sh/@supabase/supabase-js@2.39.6"; // 取得 Supabase Project URL 與 Service Role Key（僅可用於 Edge Function） const supabaseUrl = Deno.env.get("SUPABASE_URL"); const serviceRoleKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY"); if (!supabaseUrl || !serviceRoleKey) { throw new Error("Missing Supabase environment variables"); } // 用 Service Role 權限建立 Supabase client const supabase = createClient(supabaseUrl, serviceRoleKey); // 設定 CORS，允許前端跨網域呼叫 const corsHeaders = { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", "Access-Control-Allow-Credentials": "true", }; // 你所有用戶資料表（根據實際狀況補齊） const TABLES_TO_CLEAN = [ "question_status", "concept_status", "suggestions", // "profiles", // 如果你有自訂 user profile 表請加上 ]; // 刪除所有與該 user 相關的資料 async function deleteUserData(userId: string) { await Promise.all( TABLES_TO_CLEAN.map((table) => supabase.from(table).delete().eq("user_id", userId) ) ); } // Edge Function 主程式 serve(async (req: Request) => { // 處理 CORS 預檢請求 if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }); } // 僅允許 POST 請求 if (req.method !== "POST") { return new Response("Method Not Allowed", { status: 405, headers: corsHeaders }); } // 取得 Authorization header 並驗證 const token = req.headers.get("Authorization")?.replace("Bearer ", ""); if (!token) { return new Response(JSON.stringify({ error: "Unauthorized (no token)" }), { status: 401, headers: corsHeaders, }); } // 檢查該用戶是否存在於 Auth const { data, error: authError } = await supabase.auth.getUser(token); if (authError || !data?.user) { return new Response(JSON.stringify({ error: "No such account found in the server" }), { status: 404, headers: corsHeaders, }); } const userId = data.user.id; // 執行資料刪除與帳號刪除 try { await deleteUserData(userId); const { error } = await supabase.auth.admin.deleteUser(userId); if (error) { return new Response(JSON.stringify({ error: error.message }), { status: 500, headers: corsHeaders, }); } return new Response(JSON.stringify({ message: "Account and all data deleted" }), { status: 200, headers: corsHeaders, }); } catch (err) { return new Response(JSON.stringify({ error: String(err) }), { status: 500, headers: corsHeaders, }); } });
